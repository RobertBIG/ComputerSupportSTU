### 1.1. ATOMIC TYPES ----

# Numeric values (continuous)
gravity <- 9.8          # gravitational force (m/s^2)
temperature <- 27.5      # average surface temperature (°C)
radius <- 6800.3         # in kilometers
rating <- 8.9            # exploration rating (out of 10)

# Integer values (whole numbers)
moons <- 3L              # number of moons
cities <- 12L            # number of major cities
inhabitants_mil <- 45L   # number of inhabitants in millions

# Logical values (TRUE/FALSE)
has_water <- TRUE
is_inhabited <- TRUE

# Character values (text)
name <- "Aurora"
region <- "Andromeda Sector"
category <- "Class M Planet"

# Print description
cat("Planet", name, "in the", region, "is a", category,
    "with gravity", gravity, "m/s^2 and average temperature", temperature, "°C.\n")
Planet Aurora in the Andromeda Sector is a Class M Planet with gravity 9.8 m/s^2 and average temperature 27.5 °C.

### 1.3. MATRIX ----

# Create a 3x3 matrix with random numbers from -50 to 50
set.seed(42)  # for reproducibility
energy_matrix <- matrix(sample(-50:50, 9, replace = TRUE), nrow = 3, ncol = 3)

# Name rows and columns
rownames(energy_matrix) <- c("North", "Equator", "South")
colnames(energy_matrix) <- c("Zone_A", "Zone_B", "Zone_C")

# Show matrix
print(energy_matrix)
